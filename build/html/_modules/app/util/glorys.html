    def login():
        '''
        Logs current user into copernicus marine api for fetching datasets
        
        '''
        login_status = None
        try:
            retries = 5
            for i in range(0,retries):
                login_status = copernicusmarine.login(
                    username = os.getenv('CM_USERNAME'),
                    password = os.getenv('CM_PASSWORD'),
                    credentials_file =  Path("/mnt3/.copernicusmarine-credentials"),

                    )
                logger.info(f"Current login status {login_status}")
                #sleep right here
                if login_status:
                    return True
        except Exception as err:
            logger.warning(f"GLORYS Login error: {err}")
            logger.warning(f"Login status: {login_status}")

        return False

    def get_filename(self, dataset_id : str , unit_time_range : list,  variables : list,  variable_key : dict = None) -> str:
        '''
        This creates the output filename of the given dataset

        
        PARAMETERS:

        dataset_id : The unique ID of the current dataset
        unit_time_range : List of time range for the described dataset
        variable_key : (optional) A dictionary with filenaming convention substitutions for different variables

        RETURNS: 

        str : A string of the formatted output file
        
        '''
        try: 
            time_start = unit_time_range[0]
            time_end = unit_time_range[1]
            #time must be reformatted in order to create filename
            ftime_start = time_start.split("T")[0] #.strftime('%Y-%m-%d')
            ftime_end = time_end.split("T")[0] #.strftime('%Y-%m-%d')
            if variable_key:
                logger.info(f"variables: {variables}")
                logger.info(f"variable_key : {variable_key}")
                if type(variables) == list:
                    variables = list(map(variable_key.get, variables))
                    variable_string = "-".join(variables)
                else:
                    variable_string = variable_key[variables]
            else:
                if type(variables) == list:
                    variable_string = "-".join(variables)
                else:

                    variable_string = variables
            return f"GLORYS_REANALYSIS_DAILY_{dataset_id}_{variable_string}_{ftime_start}-{ftime_end}_alldays.nc"
        except Exception as err:
            logger.warning(f"Unable to create filename: {err} ")

    def get_project_dir(self, dataset_id : str, target_filepath : str = None):
        '''
        Returns the associated dataset product directory
        
        
        PARAMETERS:

        dataset_id : The unique ID of the dataset given by copernicusmarine 
        variables: Any variables collected for the dataset

        RETURNS: 

        full_output_project_path : The target relative filepath of the project directory.
        '''

        if target_filepath:
            full_output_project_path = os.path.join(target_filepath, dataset_id)
        else:
            full_output_project_path = os.path.join(self.target_filepath, dataset_id)
        try:
            if not os.path.exists(full_output_project_path):
                os.makedirs(full_output_project_path) #making full mock filepath or making dataset id? 
                logger.info(f"The Project Folder generated at: {full_output_project_path}")
                time.sleep(0.1)
            logger.info(f"Project directory located at: {full_output_project_path}")
            return full_output_project_path
        except Exception as err:
            logger.warning(f"The project folder was not able to be created: {err}")

    def fetch_dataset(self, registry_obj : Registry,  dataset_id : str|None, proportion : str, unit_time_range : list, variables : list|str, variable_key : list|str = None):
        ''''
        Fetches subset or full dataset based on constraints specified in associated object 

        
        PARAMETERS:

        registry_obj : The Registry associated with this dataset
        dataset_id : The unique identifier for the dataset given by copernicusmarine
        unit_time_range : list[2] A list containing the upper and lower time range for the dataset
        variables : Dataset variables to include in the outputted file
        variable_key : (optional) The variable key dictionary for naming datasets by predefined conventions (ex: {'sea_surface_temperature' : 'sst'})
        
        RETURNS: 

        bool : True if dataset was successfully obtained, False otherwise
        
        '''
        if isinstance(variables, list):
            vars = variables
        if isinstance(variables, str):
            vars = [variables]
        try: 

            # for nesting further folder structures, chain product_directory path to outputs from get_project__dir
            project_directory = self.get_project_dir(dataset_id = dataset_id) #creates a directory at the parent output directory
            self.project_directory = project_directory.replace(os.environ['MNT_ONE'], "")
            
            logger.info(f"this is the project directory : {self.project_directory}")
            
            if variable_key:
                subbed_variable_name = list(map(variable_key.get, vars))
                print(f"Variable name substitution: {subbed_variable_name} \n")
                project_directory = self.get_project_dir(dataset_id="_".join(subbed_variable_name), target_filepath=project_directory) #creates/gets a variable folder
                self.project_directory = project_directory.replace(os.environ['MNT_ONE'], "")
                logger.info(f"this is the project directory : {self.project_directory}")
            filename = self.get_filename(dataset_id= dataset_id, variables = variables, unit_time_range = unit_time_range, variable_key=variable_key )
        # try: 
            #with open(os.path.join(project_directory, filename), "x") as file:
            if proportion == "full":
                #copernicusmarine.get(
                dataset = copernicusmarine.open_dataset(
                    dataset_id = None,
                    dataset_version = self.dataset_id,
                    dataset_part = None,
                    #username = os.getenv('CM_USERNAME'),
                    #password = os.getenv('CM_PASSWORD'),
                    no_directories = False,
                    #output_directory = project_directory
                                    )
            if proportion == "sub":
                #copernicusmarine.subset(
                dataset = copernicusmarine.open_dataset(
                        dataset_id=self.dataset_id,
                        variables=vars,
                        minimum_longitude=self.dataset_longitude_range[0],
                        maximum_longitude=self.dataset_longitude_range[1],
                        minimum_latitude=self.dataset_latitude_range[0],
                        maximum_latitude=self.dataset_latitude_range[1],
                        start_datetime=unit_time_range[0],
                        end_datetime= unit_time_range[1],
                        #username = os.getenv('CM_USERNAME'),
                        #password = os.getenv('CM_PASSWORD'),
                        credentials_file = Path("/mnt3") / ".copernicusmarine-credentials"
                        #minimum_depth=self.dataset_depth_range[0],
                        #maximum_depth=self.dataset_depth_range[1],
                        # output_filename = filename,
                    )
            record = Registry.create_manifest_record(
                                                dataset_bytesio_obj=dataset,
                                                dataset_id = dataset_id, #param/env?
                                                dataset_date= str([unit_time_range[0].split("T")[0], unit_time_range[1].split("T")[1]]), # param
                                                date_retrieved= datetime.datetime.now().replace(microsecond=0),
                                                dataset_type= 'nc', #param
                                                dataset_source= 'GLORYS', #param
                                                number_of_records= 153,
                                                start_date=unit_time_range[0],
                                                end_date=unit_time_range[1]
                                                    )
        except Exception as e:
            logger.info(f"File could not be downloaded: {e}")  


        try: 
            retries = 5
            i = 0
            file_imported = False

            while not file_imported and i in range(0, retries): 
                logger.info(record)
                full_dataset_path = os.path.join(project_directory, filename)
                logger.info(f"Attempting file writing to: {full_dataset_path}, current number of retries: {i}")
                logger.info("Comparing to registry...\n")
                if registry_obj.update_registry(record): # if the file can be added to registry>>should be revised to be a check registry
                    if not os.path.exists(full_dataset_path): # if the file doesnt exist already 
                        file_imported = ERDDAP.write_nc(data = BytesIO(dataset.to_netcdf()), filepath = full_dataset_path)
                        i = i+1
                    else:
                        logger.warning(f"The file {filename} already exists in the location: {full_dataset_path} but is not reflected in the Registry...")
                        break
                else:
                    logger.info("This dataset has been seen \n")
                    file_imported = True
                
            logger.info("New dataset added to registry!")  
      
        except Exception as e:
            logger.info(f"There was an error: {e}")






def main():
    '''
    This main function serves as a complete example on running glorys workflows. 


    The steps outlined are:

    Logging into CopernicusMarine
    Generating a list of datasets and their identifiers: product_id, var_3d, current_time_Start, current_time_end
    Fetching the dataset
    Comparing the dataset between what has been stored
    Storing the dataset in its associated folder and adding it as a 'seen' dataset
    
    
    '''
    # Log In to CopernicusMarine for API access 
    #'''
    if GLORYS.login():
        logger.info("successful login")
    else:
        logger.info("unsuccessful login")     
    
    #'''


    # Generate a list of datasets to be imported containing the product_id, var_3d, current_time_Start, current_time_end

    datasets_to_import =  ["cmems_mod_glo_phy_my_0.083deg_P1D-m","cmems_mod_glo_phy_myint_0.083deg_P1D-m"]
    # variables = ["sea_water_potential_temperature_at_sea_floor","sea_water_potential_temperature","sea_water_salinity","ocean_mixed_layer_thickness_defined_by_sigma_theta"]
    # is_3d = [False, True, True]
    variables = ["sea_water_potential_temperature_at_sea_floor","ocean_mixed_layer_thickness_defined_by_sigma_theta"]# Not 3D 
    year_start=2021
    year_end=2021
    variable = ["sea_water_potential_temperature_at_sea_floor","sea_water_potential_temperature","sea_water_salinity","ocean_mixed_layer_thickness_defined_by_sigma_theta"]
    var_dirs = ["bottomT","temperature","salinity","mixedLayerDepth"]
    var_key = dict(zip(variable, var_dirs))
    product_id_variable_combos = [list(pair) for pair in itertools.product(datasets_to_import, variables)]
    logger.info(product_id_variable_combos)
    all_days = []
    for y in range(year_start,year_end+1):
        dt = datetime.datetime(y,1,1)
        end = datetime.datetime(y,5,30)
        step = datetime.timedelta(days = 1)
        #all_days = [] # comment out if you range all days across all years in one array
        while dt <= end:
            all_days.append(dt.strftime('%Y-%m-%d'))
            dt += step
    #makes a formatted list of timeranges from all days
    daily_time_ranges = list(map(lambda x : [x+"T00:00:00",x+"T23:59:59"], all_days))
    combinations = list(itertools.product(product_id_variable_combos, daily_time_ranges))
    logger.info(combinations[1:10])
    # returns a list of tuples where each tuple is ([prod id, variable], [timestart, timeend]) or [ (['cmems_mod_glo_phy_my_0.083deg_P1D-m', 'sea_water_potential_temperature_at_sea_floor'], ['2021-01-02T00:00:00', '2021-01-02T23:59:59'])
    min_lon = -82.5
    max_lon = -51.5
    min_lat = 22.5
    max_lat = 48.5
    # For usage in testing random combinations of datasets
    random_combinations = [combinations[random.randint(0, len(combinations))] for x in range(0, 10)]
    #combinations = random_combinations

    #Iterate over all datasets with the expected parameters (product_id, var_3d, current_time_Start, current_time_end), retrieve, validate, and record
    for i in range(len(combinations[0:10])):
 
        product_id = combinations[i][0][0]
        var_3d = combinations[i][0][1]
        current_time_start = combinations[i][1][0]
        current_time_end = combinations[i][1][1] 

        logger.info(f"Current variables: {var_3d}")
        logger.info(
            f'''
        product_id: {product_id}\n
        currrent_variable: {var_3d}\n
        time_start: {current_time_start} \n
        time_end: {current_time_end}
        '''
        )
        # The constraints of the glorys object will be provided by the config file
        glorys_engine = GLORYS(
            dataset_id=product_id, 
            dataset_date_range = "",
            dataset_latitude_range = [min_lat, max_lat], 
            dataset_longitude_range = [min_lon, max_lon], 
            dataset_depth_range = None,
            input_id = "", 
            variables = "",
            time_range = []
            )
        # Fetch dataset against copernicusmarine API
        glorys_engine.fetch_dataset(
            proportion = "sub",
            dataset_id=product_id,
            unit_time_range=[current_time_start, current_time_end],
            variables=var_3d,
            variable_key=var_key
        )
