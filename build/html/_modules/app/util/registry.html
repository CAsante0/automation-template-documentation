    def _create_checksum(dataset : BytesIO | xr.Dataset, dataset_filepath : str = None, filetype : str = "nc" ):
        '''
        Create a checksum_id, this will be searched within the registry. This can currently only handle netcdf datatypes

        PARAMETERS:

        dataset : the full dataset imported expressed as an xr.Dataset or BytesIO object
        
        '''
        # Load dataset if BytesIO, otherwise assume xr.Dataset
        if isinstance(dataset, xr.Dataset):
            ds = dataset
        elif isinstance(dataset, BytesIO):
            dataset.seek(0)
            ds = dataset
        else:
            raise ValueError("dataset must be an xarray.Dataset or a BytesIO object")

 
        volatile_var_attrs = ['history', 'creation_date', 'source', 'comment']
        for var in ds.data_vars:
            for attr in volatile_var_attrs:
                ds[var].attrs.pop(attr, None)
            # Sort variable attrs for consistent ordering
            ds[var].attrs = collections.OrderedDict(sorted(ds[var].attrs.items()))

        # Sort global attributes for consistency
        ds.attrs = collections.OrderedDict(sorted(ds.attrs.items()))

        # Sort variables and coordinates alphabetically
        vars_sorted = sorted(ds.data_vars)
        coords_sorted = sorted(ds.coords)

        hasher = hashlib.sha256()

        # hash global attri
        for k, v in ds.attrs.items():
            hasher.update(k.encode('utf-8'))
            hasher.update(str(v).encode('utf-8'))

        # hash coordinate names
        for coord in coords_sorted:
            hasher.update(coord.encode('utf-8'))
            coord_data = ds.coords[coord].values
            hasher.update(coord_data.tobytes())

        # hashed variable names, their data, and sorted attributes
        for var in vars_sorted:
            hasher.update(var.encode('utf-8'))
            var_data = ds[var].values

            # Round floats to avoid tiny floating point noise causing different hashes
            if np.issubdtype(var_data.dtype, np.floating):
                var_data = np.round(var_data, decimals=6)

            hasher.update(var_data.tobytes())

            for k, v in ds[var].attrs.items():
                hasher.update(k.encode('utf-8'))
                hasher.update(str(v).encode('utf-8'))

        return hasher.hexdigest()

    def _create_hash_record(dataset_name : str, dataset_date : str, dataset_num_observations : str):
        '''
        This creates a hash record which acts as a unique identifier for fetched datasets.

        PARAMETERS:
        dataset_name : The full filename of the dataset
        dataset_date : The temporal range of the dataset expressed as a day, month, season, year, or a list ex: [2020, 2025]

        RETURNS: 
        A hash record of the dataset
        '''
        hash_string = "_".join([dataset_name, dataset_date, dataset_num_observations])
        encoded_hash_string = str.encode(hash_string)
        record_hash_code = hashlib.sha256(encoded_hash_string)

        return record_hash_code

    def create_registry(registry_folder_filepath : str, registry_id : str) -> None: 
        '''
        Initializes a workflow registry in logging folder for the defined dataset/workflow.

        
        PARAMETERS:

        manifest_id : The unique ID of a workflow manifest registry

        RETURNS: 

        boolean : True if the registry was created at the location provided and False otherwise
        '''
        reg = f"{registry_folder_filepath}\\{registry_id}*"
        logger.info(reg)
        matching_files = glob.glob(reg)
        logger.info(matching_files)
        if matching_files:
            logger.info(f"A matching registry has been located at {matching_files[0]}")
            return matching_files[0]
        try:
            registry_dict = {
                "checksum_id": [],
                "dataset_id" : [],
                "dataset_date" :[],
                "data_start_date" : [],
                "data_end_date" : [],
                "date_retrieved" : [],
                "dataset_type" : [],
                "dataset_source" : [],
                "number_of_records" : [],
                "filepath" : [],
                'action': []
            }
            df = pd.DataFrame(registry_dict)

            # create registry name
            registry_filename = f"{registry_id}_{datetime.now().strftime("%Y-%m-%d")}" #this should be updated to include a timestamp of the date created
            registry_filepath = os.path.join(registry_folder_filepath,registry_filename) + ".csv"
            
            if not os.path.exists(registry_filepath):
                df.to_csv(registry_filepath, index=0)
                logger.info(f"A file has been created at: {registry_filepath}")
                return registry_filepath
        except: 
            logger.info(f"the registry couldnt be created at the location: {registry_filepath}")
        return None
