def config_file_handler(filename : str, validation_config : str = None) -> dict:
    """
    Function for handling all configuration parsing.


    PARAMETERS: 

    filename : the relative path of the configuration file

    RETURNS:

    param_dict : a dictionary of the parsed configuration.yaml
    """
    config_arg_object = ParseConfig(config_path=filename, validation_config_path=validation_config)
    if config_arg_object.validate_config_params():
        return config_arg_object.parameter_args_dict
    raise logger.info("Your configuration file is malformed. ")



def name_files(type : str) -> None:
    ''''
    Temporary function for defining names for files, can be reformatted such that new file names only need to be added based on data/class type.

    PARAMETERS:

    type : The filetype outputted
    
    RETURNS:
     
    str : composed filename of dataset with appropriate file type.
    
    '''

    if type =='csv':
        return f"tabledap_{time.strftime("%Y-%m-%d", time.localtime())}.csv"
    if type == 'nc':
        return f"griddap_{time.strftime("%Y-%m-%d", time.localtime())}.nc"


def main_erddap(input_config : str, validation_config :str):
    '''
    Initialize an ERDDAP Import Workflow 

    Parses the User Provided configuration files
    Creates an ERDDAP Object
    Fetches the target resource and saves it to the ERDDAP Object
    
    '''
    
    # Read arguments
    source_config = input_config
    validation_config = validation_config


    logger.info("Grabbing config values...")
    param_dict = config_file_handler(filename = source_config, validation_config = validation_config) 
   
    # Setting parameter dictionaries
    dataset_params = param_dict['datasets'] # datasets to search for 
    server_params = param_dict['server'] # source/target endpoints 
    server_params.update(param_dict['target_server']) #updates with newly added target_server field, overwrites variables shared with source_server fields
    query_params = param_dict['query'] # file details
    file_output_type = query_params['output_format']

    workflow = param_dict['workflow']['service_name']
    logger.info(workflow)

    logger.info(f"dataset_params: {dataset_params}")
    logger.info(f"server_params: {server_params}")
    logger.info(f"query_params: {query_params}")
    dataset_id = dataset_params['name']

    # Create ERDDAP Object
    import_ds_obj = ERDDAP(
            base_erddap_url=server_params['url'],
            dataset_id=dataset_id,
            dap_type=server_params['dap_type'],
            file_type=query_params['output_format'],
            target_host=os.getenv("MNT_ONE"),
            param_variables= dataset_params['variables'][0]['name'],
            constraint_variables=dataset_params['variables'][0]['range']
    )
    # Fetching Dataset with provided configuration file arguments
    dataset = import_ds_obj.fetch_dataset(file_output_type, stream=False)

    logger.info("Outputting data to target server")
    registry = Registry(
                registry_folder_filepath= os.path.join(os.getenv("LOGGER_DIR"), "registry"),
                registry_id=dataset_id,
                resource_filepath=os.getenv("MNT_ONE"),
    )
    logger.info(file_output_type)
    ERDDAP.export_dataset(
        registry_obj = registry,
                dataset_id = dataset_id,
                filepath = os.getenv("MNT_ONE"), 
                filename =  name_files(file_output_type),
                file_type = file_output_type , 
                dataset = dataset)
    
    # Log successful file export
    logger.info("Completed!")
    
    try: 
        r = requests.post(os.getenv("SERVICE_NAME"), data = {'animals' : ['seal', 'fish', 'whale']})
        logger.info(r.json())
    except requests.exceptions.RequestException as e:
        time.sleep(2)

def check_monthly(current_registry : str, current_month : str, current_year : str) -> bool: 
    '''
    Checks registry of workflow to ensure full month (each day) is stored before running monthly workflows.
    
    
    PARAMETERS:

    current_registry : The registry of the current resource
    current_month : The month being identified
    current_year : The year being identified

    RETURNS:

    boolean, list(optional) True if all datasets in a given month exist, False with list of missing dates otherwise
    
    '''
    registry = pd.read_csv(current_registry)

    def get_date_only(date_string):
        try:
            date_parts = ast.literal_eval(date_string)  # safely parses string like "['2025-07-01', '12:00:00']"
            return date_parts[0]  # return the date part only
        except Exception:
            return None  # skip rows that are malformed
        
    registry['date'] = registry['dataset_date'].apply(get_date_only)
    #valid_rows = registry[(registry['status'] == 'imported') & (registry['date'].notnull())] #for after the inclusion of a row that 
    valid_rows = registry['date'].dropna().unique()
    total_days = monthrange(current_year, current_month)[1]
    expected_dates = {
        datetime.datetime(current_year, current_month, day).strftime("%Y-%m-%d")
        for day in range(1, total_days + 1)
    }
    imported_dates = set(valid_rows)
    missing_dates = sorted(expected_dates - imported_dates)
    is_ready = len(missing_dates) == 0 # checks that there are no missing dates
    return is_ready, missing_dates

def generate_glorys_combos_yearly(datasets_to_import : list = None, target_variables : list = None, year_start : int = 1997, year_end : int = 2020, product_ids : list = None):
    '''
    Create a list of yearly glorys datasets for import based on the dot product

    PARAMETERS
    datasets_to_import/product_ids : A list[str] of product IDs of glorys datasets
    target_variables : A list[str] of variables to be imported based on the product ID
    var_key : (optional) A dictionary to define naming conventions of a given dataset
    year_start : The start of the year for the given dataset range 
    year_end : The terminal year of the datasets range

    RETURNS
    A list of combinations of product_ids * target_variables *
    # returns a list of tuples where each tuple is ([prod id, variable], [timestart, timeend]) or [ (['cmems_mod_glo_phy_my_0.083deg_P1D-m', 'sea_water_potential_temperature_at_sea_floor'], ['2021-01-02T00:00:00', '2021-01-02T23:59:59'])
    '''

    datasets_to_import =  ["cmems_mod_glo_phy_my_0.083deg_P1D-m","cmems_mod_glo_phy_myint_0.083deg_P1D-m"]
    #target_variables = ["sea_water_potential_temperature_at_sea_floor","ocean_mixed_layer_thickness_defined_by_sigma_theta"]
    target_variables = ["sea_water_potential_temperature","sea_water_salinity"]
    years=range(year_start,year_end+1) 

    product_id_variable_combos = [list(pair) for pair in itertools.product(datasets_to_import, target_variables)]
    logger.info(product_id_variable_combos)
    
    
    all_years = []
    for year in years: 
        date_start = datetime.datetime(year, 1, 1).strftime("%Y-%m-%d")+"T00:00:00"
        date_end = datetime.datetime(year, 12, 31).strftime("%Y-%m-%d")+"T23:59:59"
        all_years += [[date_start, date_end]]
    #daily_time_ranges = list(map(lambda x : [x+"T00:00:00",x+"T23:59:59"], all_days))
    
    print(all_years)
    combinations = list(itertools.product(product_id_variable_combos, all_years))
    logger.info(f"Sample of generated yearly combinations: {combinations[1:10]}")
    # returns a list of tuples where each tuple is ([prod id, variable], [timestart, timeend]) or [ (['cmems_mod_glo_phy_my_0.083deg_P1D-m', 'sea_water_potential_temperature_at_sea_floor'], ['2021-01-02T00:00:00', '2021-01-02T23:59:59'])
    return combinations


def generate_glorys_combinations_daily(datasets_to_import : list = None, target_variables : list = None, year_start : int = 2021, year_end : int = 2021, product_ids : list = None):
    '''
    Create a list of daily glorys datasets for import based on the dot product

    PARAMETERS
    datasets_to_import/product_ids : A list[str] of product IDs of glorys datasets
    target_variables : A list[str] of variables to be imported based on the product ID
    var_key : (optional) A dictionary to define naming conventions of a given dataset
    year_start : The start of the year for the given dataset range 
    year_end : The terminal year of the datasets range

    RETURNS
    A list of combinations of product_ids * target_variables *
    # returns a list of tuples where each tuple is ([prod id, variable], [timestart, timeend]) or [ (['cmems_mod_glo_phy_my_0.083deg_P1D-m', 'sea_water_potential_temperature_at_sea_floor'], ['2021-01-02T00:00:00', '2021-01-02T23:59:59'])
    '''

    datasets_to_import =  ["cmems_mod_glo_phy_my_0.083deg_P1D-m","cmems_mod_glo_phy_myint_0.083deg_P1D-m"]
    #target_variables = ["sea_water_potential_temperature","sea_water_salinity"]
    #variables = ["sea_water_potential_temperature_at_sea_floor","ocean_mixed_layer_thickness_defined_by_sigma_theta"]     # 2D yearly variables
    target_variables = ["sea_water_potential_temperature_at_sea_floor","ocean_mixed_layer_thickness_defined_by_sigma_theta"]
    product_id_variable_combos = [list(pair) for pair in itertools.product(datasets_to_import, target_variables)]
    logger.info(product_id_variable_combos)
    
    #daily : 3D 
    all_days = []
    for y in range(year_start,year_end+1):
        dt = datetime.datetime(y,1,1)
        end = datetime.datetime(y,5,30)
        step = datetime.timedelta(days = 1)
        #all_days = [] # comment out if you range all days across all years in one array
        while dt <= end:
            all_days.append(dt.strftime('%Y-%m-%d'))
            dt += step
    #makes a formatted list of timeranges from all days
    daily_time_ranges = list(map(lambda x : [x+"T00:00:00",x+"T23:59:59"], all_days))
    
    combinations = list(itertools.product(product_id_variable_combos, daily_time_ranges))
    logger.info(f"Sample of generated daily combinations: {combinations[1:10]}")

    return combinations
